 SELECT A.MaChamCong EMPLOYEE_CODE
      ,A.NgayCham WORKINGDAY
      ,A.GioCham IN_OUT_VALUE
      ,A.KieuCham CHECKTYPE
      ,A.MaSoMay MACHINE_CODE
      ,CONVERT(char, A.GioCham, 103) + ' ' + CONVERT(char, A.GioCham, 108) as IN_OUT_VALUE1
  FROM CheckInOut A
  WHERE convert(char, A.GioCham, 112) >='{0}'
  AND  convert(char, A.GioCham, 112) <='{1}'
  ORDER BY A.MaChamCong
